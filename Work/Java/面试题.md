String s1 = "abc";
String s2 = "a" + "b" + "c"; //编译优化机制会将 "a" + "b" + "c" 变为 "abc"

String s1 = "abc";
String s2 = "ab";
String s3 = s2 + "c"; // 编译阶段无法确定s2的值
System.out.println(s1 == s3); // false
